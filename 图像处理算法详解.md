## 13. 撤销/重做功能
### 算法原理
1. **栈结构**：使用两个栈(undo_stack和redo_stack)保存历史状态
2. **差异存储**：为节省内存，存储图像差异而非完整图像
3. **操作边界**：限制历史记录数量防止内存耗尽

### 实现步骤与代码解释
1. **基础栈实现**  
   使用双栈结构管理历史状态：
   ```python
   class HistoryManager:
       def __init__(self, max_history=20):
           self.undo_stack = []  # 撤销栈
           self.redo_stack = []  # 重做栈
           self.max_history = max_history  # 最大历史记录数
           self.current_image = None  # 当前图像
   ```

2. **存储优化实现**  
   使用图像差异减少内存占用：
   ```python
   def _get_image_diff(self, old_img, new_img):
       """计算图像差异矩阵"""
       if old_img is None or new_img is None:
           return None
       return new_img.astype(np.int16) - old_img.astype(np.int16)
       
   def _apply_diff(self, base_img, diff):
       """应用差异到基础图像"""
       if base_img is None or diff is None:
           return None
       result = base_img.astype(np.int16) + diff
       return np.clip(result, 0, 255).astype(np.uint8)
   ```

3. **核心操作方法**  
   实现撤销/重做逻辑：
   ```python
   def push_state(self, new_image):
       """压入新状态"""
       if self.current_image is not None:
           diff = self._get_image_diff(self.current_image, new_image)
           self.undo_stack.append((self.current_image.copy(), diff))
           
           # 限制历史记录数量
           if len(self.undo_stack) > self.max_history:
               self.undo_stack.pop(0)
               
       self.current_image = new_image.copy()
       self.redo_stack = []  # 新操作清空重做栈
   
   def undo(self):
       """撤销操作"""
       if self.undo_stack:
           prev_image, diff = self.undo_stack.pop()
           self.redo_stack.append((self.current_image.copy(), 
                                 self._get_image_diff(prev_image, self.current_image)))
           self.current_image = prev_image.copy()
           return self.current_image
       return None
   
   def redo(self):
       """重做操作"""
       if self.redo_stack:
           next_image, diff = self.redo_stack.pop()
           self.undo_stack.append((self.current_image.copy(), 
                                 self._get_image_diff(self.current_image, next_image)))
           self.current_image = next_image.copy()
           return self.current_image
       return None
   ```

5. **内存优化策略**  
   定期清理和压缩历史记录：
   ```python
   def optimize_memory(self):
       """优化内存使用"""
       if len(self.undo_stack) > self.max_history * 0.8:
           # 保留最近的50%记录
           keep = len(self.undo_stack) // 2
           self.undo_stack = self.undo_stack[-keep:]
           
       # 压缩差异数据
       for i in range(len(self.undo_stack)):
           _, diff = self.undo_stack[i]
           if diff is not None:
               self.undo_stack[i] = (None, diff.astype(np.int8))  # 使用int8节省空间
   ```

完整类实现：
```python
class AdvancedHistoryManager:
    """带内存优化的历史管理器"""
    def __init__(self, max_history=20, compress_after=10):
        self.undo_stack = []
        self.redo_stack = []
        self.max_history = max_history
        self.compress_after = compress_after
        self.operation_count = 0
        self.current_image = None
        
    def push_state(self, new_image):
        """记录新状态"""
        if self.current_image is None:
            self.current_image = new_image.copy()
            return
            
        diff = self._get_image_diff(self.current_image, new_image)
        self.undo_stack.append((self.current_image.copy(), diff))
        self.current_image = new_image.copy()
        self.redo_stack = []
        
        # 内存管理
        self.operation_count += 1
        if self.operation_count >= self.compress_after:
            self.optimize_memory()
            self.operation_count = 0
            
        # 限制历史记录数量
        if len(self.undo_stack) > self.max_history:
            self.undo_stack.pop(0)
    
    def undo(self):
        """撤销操作"""
        if not self.undo_stack:
            return None
            
        prev_image, diff = self.undo_stack.pop()
        if prev_image is None:  # 如果是压缩后的记录
            prev_image = self._apply_diff(self.current_image, -diff)
            
        self.redo_stack.append((self.current_image.copy(), 
                              self._get_image_diff(prev_image, self.current_image)))
        self.current_image = prev_image.copy()
        return self.current_image
    
    def redo(self):
        """重做操作"""
        if not self.redo_stack:
            return None
            
        next_image, diff = self.redo_stack.pop()
        if next_image is None:  # 如果是压缩后的记录
            next_image = self._apply_diff(self.current_image, diff)
            
        self.undo_stack.append((self.current_image.copy(), 
                              self._get_image_diff(self.current_image, next_image)))
        self.current_image = next_image.copy()
        return self.current_image
    
    def _get_image_diff(self, old_img, new_img):
        """计算图像差异"""
        if old_img.shape != new_img.shape:
            return None
        return new_img.astype(np.int16) - old_img.astype(np.int16)
    
    def _apply_diff(self, base_img, diff):
        """应用差异"""
        if diff is None:
            return base_img.copy()
        result = base_img.astype(np.int16) + diff
        return np.clip(result, 0, 255).astype(np.uint8)
    
    def optimize_memory(self):
        """优化内存使用"""
        # 压缩undo栈
        compressed = []
        for i, (img, diff) in enumerate(self.undo_stack):
            if i % 2 == 0 or i == len(self.undo_stack)-1:  # 保留部分完整图像
                compressed.append((img, diff))
            else:
                compressed.append((None, diff.astype(np.int8)))  # 只保留差异
        self.undo_stack = compressed
        
        # 限制栈大小
        if len(self.undo_stack) > self.max_history:
            self.undo_stack = self.undo_stack[-self.max_history:]
        
        # 清空redo栈
        self.redo_stack = []
    
    def clear(self):
        """清空历史记录"""
        self.undo_stack = []
        self.redo_stack = []
        self.current_image = None
        self.operation_count = 0
